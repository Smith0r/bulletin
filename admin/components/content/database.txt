CREATE TABLE `#__articles` (`id` int(11) NOT NULL AUTO_INCREMENT, `title` varchar(500) DEFAULT NULL, `alias` varchar(500) NOT NULL, `category_id` int(11) NOT NULL DEFAULT '0', `content` text, `published` int(11) NOT NULL DEFAULT '1', `author_id` int(11) DEFAULT NULL, `hits` int(11) DEFAULT NULL, `meta_description` varchar(500) DEFAULT NULL, `image` varchar(1000) DEFAULT NULL, `publish_time` int(11) DEFAULT NULL, PRIMARY KEY (`id`));
